```python
n = int(input())
memo = [0]*(n+1)

def go(n):
    if n == 1:
        return 0
    elif memo[n] > 0:
        return memo[n]
    else:
        memo[n] = go(n-1) + 1
        if n%2 == 0:
            temp = go(n//2)+1
            if memo[n] > temp:
                memo[n] = temp
        if n%3 == 0:
            temp = go(n//3)+1
            if memo[n] > temp:
                memo[n] = temp
        return memo[n]

print(go(n))

```

```python
    memo[i] = memo[i-1] + 1
    if i % 2 == 0:
        temp = memo[i // 2] + 1
        if memo[i] > temp:
            memo[i] = temp

    if i % 3 == 0:
        temp = memo[i // 3] + 1
        if memo[i] > temp:
            memo[i] = temp
```
